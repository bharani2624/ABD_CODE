#include <stdio.h>
int main(){
    int n,i;
    scanf("%d",&n);
    if(n%2==0){
        for(i=2;i<n;i++){
        if(i%2==0){
            printf("%d ",i);
        }
    }
    }
    else{
        for(i=1;i<n;i++){
            if(i%2!=0){
                printf("%d ",i);
            }
        }
    }
}
